font-style-bold = default
font-style-italic = default
font-style-bold-italic = default
# リクエストされたスタイルが指定されたフォントファミリで利用できない場合、
# Ghosttyが自動的にスタイルを生成すべきかを制御します。
#
# Ghosttyはフォントに特定のスタイルがない場合、太字、斜体、太字斜体の
# スタイルを自動生成できます。太字の場合は、グリフの周囲に異なる厚さで
# アウトラインを描画することで実現されます。斜体の場合はグリフに傾きを
# 適用することで実現されます。太字斜体の場合はこれら両方が適用されます。
#
# 自動生成されたスタイルは完璧ではなく、一般的にはそのスタイルが
# ネイティブに含まれているフォントほど見栄えが良くありません。
# しかし、フォントにスタイルがない場合でもスタイル付きテキストを
# 提供する場合に役立ちます。
#
# これを「false」または「true」に設定して、自動生成されたスタイルを
# 完全に無効化または有効化できます。「no-bold」、「no-italic」、
# 「no-bold-italic」を使用して特定のスタイルのみを無効化できます。
# 複数のスタイルを無効化する場合はカンマで区切ります。例えば、
# 「no-bold,no-italic」のようにします。
#
# 利用可能なスタイルキーは: `bold`、`italic`、`bold-italic`。
#
# 自動生成されたスタイルが無効化されている場合、リクエストされた
# スタイルが利用できなければ通常のスタイルが代わりに使用されます。
# フォントがリクエストされたスタイルを持っている場合、そのスタイルは
# 自動生成ではないため、フォントはそのまま使用されます。
#
# 警告: `bold`または`italic`を無効化するが`bold-italic`は無効化しないという
# 簡単なミスをすることがあります。`bold`または`italic`のみを無効化しても、
# `bold-italic`スタイル内の両者を無効化することはできません。
# `bold-italic`を無効化する場合は、明示的に無効化する必要があります。
# `bold-italic`を部分的に無効化することはできません。
#
# デフォルトでは、自動生成されたスタイルは有効化されています。
font-synthetic-style = bold,italic,bold-italic
