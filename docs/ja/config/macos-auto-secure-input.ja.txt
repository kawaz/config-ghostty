# true に設定されている場合、macOS 上の Ghostty はパスワードプロンプト
# が表示されていることを検出したときに、「セキュアインプット」機能
# を自動的に有効にします。
#
# 「セキュアインプット」は macOS のセキュリティ機能であり、アプリケーション
# がキーボードイベントを読むことを防止します。これは常に
# `Ghostty > Secure Keyboard Entry` メニューアイテムを使用して
# 手動で有効にすることができます。
#
# 自動パスワードプロンプト検出はヒューリスティックに基づいており、
# 常に期待通りに動作するとは限りません。特に SSH 接続では機能しませんが、
# 他にも動作しない場合がある可能性があります。
#
# この機能を無効にする理由としては、正当なアクセシビリティソフトウェア
#（またはアクセシビリティ API を使用するソフトウェア）に干渉している
# ことが判明した場合が考えられます。セキュアインプットはどのアプリケーション
# がキーボードイベントを読むことも防止するためです。
macos-auto-secure-input = true
