window-position-y =
# ウィンドウ状態の保存と復元を有効にするかどうか。ウィンドウ状態には、
# ウィンドウの位置、サイズ、タブ、分割など が含まれます。作業ディレクトリの保存など
# 一部のウィンドウ状態にはシェル統合が必要です。詳細は `shell-integration` を参照してください。
#
# この設定には 3 つの有効な値があります：
#
#   * `default` はデフォルトのシステム動作を使用します。macOS では、アプリケーションが
#     強制終了された場合、または Settings.app でシステム全体で設定されている場合にのみ、
#     状態を保存します。
#
#   * `never` はウィンドウ状態を保存しません。
#
#   * `always` は Ghostty の終了時に常にウィンドウ状態を保存します。
#
# Ghostty が実行されていない状態でこの値を `never` に変更した場合、次の Ghostty の
# 起動時はウィンドウ状態を復元しません。
#
# Ghostty が実行されていない状態でこの値を `default` に変更し、前回の終了時に状態が
# 保存されていた場合、次の Ghostty の起動時は依然としてウィンドウ状態を復元します。
# これは、Ghostty が前回の終了が強制保存によるものかどうかを判断できないためです
# (macOS はこの情報を提供しません)。
#
# Ghostty が実行されていない状態でこの値を変更してウィンドウ状態が保存されるようにした場合、
# Ghostty は有効な場合にのみ終了時に状態を保存するため、前のウィンドウ状態は復元されません。
#
# デフォルト値は `default` です。
#
# 現在、このオプションは macOS でのみサポートされています。Linux では効果がありません。
window-save-state = default
