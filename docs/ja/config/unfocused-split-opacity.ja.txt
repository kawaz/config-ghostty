# フォーカスされていないスプリットの不透明度（透明性の反対）。
# フォーカスされていないスプリットはデフォルトでわずかに薄くなり、
# どのスプリットがフォーカスされているかを見やすくします。この機能を無効にするには、
# この値を1に設定します。
#
# 値1は完全に不透明で、値0は完全に透明です。"0"は有用ではないため
# （ウィンドウを非常に奇妙に見せる）、最小値は0.15です。この値は
# まだ奇妙に見えますが、少なくとも何が起きているかは見えます。
# 0.15～1の範囲外の値は、最も近い有効な値にクリップされます。
unfocused-split-opacity = 0.7
