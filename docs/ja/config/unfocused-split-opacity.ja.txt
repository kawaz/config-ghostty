# フォーカスされていないスプリットの不透明度レベル（透明度の反対）。
# フォーカスされていないスプリットはデフォルトではわずかにフェードアウトされており、
# どのスプリットがフォーカスされているかを簡単に確認できるようになっている。
# この機能を無効にするには、この値を 1 に設定する。
#
# 1 の値は完全に不透明であり、0 の値は完全に透明である。
# 「0」は有用ではない（ウィンドウが非常に奇妙に見える）ため、
# 最小値は 0.15 である。この値はまだ奇妙に見えるが、少なくとも
# 何が起こっているのかを見ることができる。0.15 から 1 の範囲外の値は
# 最も近い有効な値にクランプされる。
unfocused-split-opacity = 0.7
