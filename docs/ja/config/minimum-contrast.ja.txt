# 前景色と背景色間の最小コントラスト比。
# コントラスト比は1から21の間の値です。1の値では
# コントラストなし（例：黒背景に黒）が許可されます。この値は
# [WCAG 2.0仕様](https://www.w3.org/TR/WCAG20/)で定義されている
# コントラスト比です。
#
# 見えなくなったテキスト（背景と同じ色）を避けたい場合は、
# 1.1の値が良いです。読みにくいテキストを避けたい場合は、
# 3以上の値が良いです。値が高いほど、テキストが黒または白に
# なる可能性が高くなります。
#
# この値は絵文字または画像には適用されません。
minimum-contrast = 1
