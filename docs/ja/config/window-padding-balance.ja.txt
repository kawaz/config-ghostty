# ビューポートの寸法は通常、セルサイズで完全に割り切れません。この場合、
# 列の終端と最終行の下部に余分なパディングが存在する可能性があります。
# これが `true` の場合、この余分なパディングは4つの辺すべてに自動的に
# バランスさせられ、片側の不均衡を最小化します。これが `false` の場合、
# 左上のグリッドセルは常に辺に密着し、他の `window-padding` オプションで
# 指定されるパディング以外にはパディングはありません。
#
# 他の `window-padding` フィールドが設定されており、これが `true` の場合でも、
# 適用されます。他のパディングが最初に適用され、実際に存在するグリッドセルの
# 数に影響を与える可能性があり、これは最後に適用されて、特定のビューポート
# サイズとグリッドセルサイズに基づいてパディングをバランスさせます。
window-padding-balance = false
