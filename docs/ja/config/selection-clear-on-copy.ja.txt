# コピー後に選択テキストをクリアするかどうか。デフォルトは `false` です。
#
# `true` に設定すると、`copy_to_clipboard` キーバインディングを呼び出す
# コピー操作の後、選択状態が自動的にクリアされます。
# 重要なことに、`copy-on-select` 経由でコピー操作が呼び出された場合は、
# 選択状態はクリアされません。
#
# `false` に設定すると、コピー後も選択状態が表示されたままになり、
# コピーされた内容を確認でき、同じ選択状態に対して追加の操作を
# 実行できます。
selection-clear-on-copy = false
